<script>
  // Debounce helper
  function debounce(fn, wait){
    let t;
    return function(...args){
      clearTimeout(t);
      t = setTimeout(() => fn.apply(this, args), wait);
    };
  }

  // Search/filter logic
  (function(){
    const input = document.getElementById('search');
    const clearBtn = document.getElementById('clearSearch');
    const tbody = document.querySelector('#productsTable tbody');
    const rows = Array.from(tbody.querySelectorAll('tr'));

    function normalize(s){
      return (s||'').toString().trim().toLowerCase();
    }

    function filterRows(){
      const q = normalize(input.value);
      if(!q){
        rows.forEach(r => { r.classList.remove('hidden'); });
        return;
      }
      rows.forEach(r => {
        const name = normalize(r.dataset.name);
        const sku = normalize(r.dataset.sku);
        const meta = normalize(r.dataset.meta);
        const hay = name + ' ' + sku + ' ' + meta;
        if(hay.includes(q)){
          r.classList.remove('hidden');
        } else {
          r.classList.add('hidden');
        }
      });
    }

    const debouncedFilter = debounce(filterRows, 180);

    input.addEventListener('input', debouncedFilter);
    clearBtn.addEventListener('click', function(e){
      e.preventDefault();
      input.value = '';
      input.focus();
      filterRows();
    });

    // support Enter to focus first visible action (optional)
    input.addEventListener('keydown', function(e){
      if(e.key === 'Enter'){
        e.preventDefault();
        const firstVisible = rows.find(r => !r.classList.contains('hidden'));
        if(firstVisible){
          const edit = firstVisible.querySelector('a.btn-action.edit');
          if(edit) edit.focus();
        }
      }
    });
  })();
</script>
